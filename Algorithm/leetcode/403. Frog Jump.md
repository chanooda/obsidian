function canCross(stones: number[]): boolean {
    const n = stones.length;

    // 돌의 위치를 빠르게 찾기 위한 맵
    const stoneSet = new Set(stones);

    // 메모이제이션을 위한 Map
    const memo: Map<string, boolean> = new Map();

    const dfs = (pos: number, k: number): boolean => {
        const key = `${pos},${k}`;
        if (memo.has(key)){
            console.log(key);
            console.log(memo.get(key))
            return memo.get(key)!;
        } 
        

        // 마지막 돌에 도달했는지 확인
        if (pos === stones[n - 1]) return true;

        // 점프 가능한 거리
        for (let step = k - 1; step <= k + 1; step++) {
            if (step > 0 && stoneSet.has(pos + step)) {
                if (dfs(pos + step, step)) {
                    memo.set(key, true);
                    return true;
                }
            }
        }

        memo.set(key, false);
        return false;
    };

    // 첫 번째 점프는 항상 1
    return dfs(stones[0], 0);
}
